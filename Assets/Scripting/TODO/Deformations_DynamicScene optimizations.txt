float opCheapBend( in sdf3d primitive, in vec3 p )
{
    const float k = 10.0; // or some other amount
    float c = cos(k*p.x);
    float s = sin(k*p.x);
    mat2  m = mat2(c,-s,s,c);
    vec3  q = vec3(m*p.xy,p.z);
    return primitive(q);
}

// Octahedron - exact
float sdOctahedron( vec3 p, float s )
{
    p = abs(p);
    float m = p.x+p.y+p.z-s;
    vec3 q;
        if( 3.0*p.x < m ) q = p.xyz;
    else if( 3.0*p.y < m ) q = p.yzx;
    else if( 3.0*p.z < m ) q = p.zxy;
    else return m*0.57735027;
    
    float k = clamp(0.5*(q.z-q.y+s),0.0,s); 
    return length(vec3(q.x,q.y-s+k,q.z-k)); 
}

float Octahedron_SDF( float3 position, Strike strike) // not exact
{
    float3 p = abs(position-strike.position);
    return (p.x+p.y+p.z- strike.width )*0.57735027;
}

    float d2 = displacement(p);
    return d1+d2;
}

float Intersection_SDF( float domain, float evaluate ) {  return max(domain,evaluate); }
float Xor_SDF( float domain, float evaluate ) { return max(min(domain,evaluate),-max(domain,evaluate)); }

float Round_SDF( in float evaluate, in float rad ) { return evaluate - rad; }

float SmoothUnion_SDF( float domain, float evaluate, float k )
{
    k *= 4.0;
    float h = max(k-abs(evaluate-domain),0.0);
    return min(evaluate, domain) - h*h*0.25/k;
}
float SmoothSubtraction_SDF( float domain, float evaluate, float k )
{
    //k *= 4.0;
    // float h = max(k-abs(-evaluate-domain),0.0);
    // return max(-evaluate, domain) + h*h*0.25/k;
    return -SmoothUnion_SDF(domain,-evaluate,k);
}
float SmoothIntersection_SDF( float domain, float evaluate, float k )
{
    //k *= 4.0;
    // float h = max(k-abs(d1-d2),0.0);
    // return max(d1, d2) + h*h*0.25/k;
    return -SmoothUnion_SDF(-domain,-evaluate,k);
}

}
